## Install

```bash
npm install --save burokku
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'burokku'
import 'burokku/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
